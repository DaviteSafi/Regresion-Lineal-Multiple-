# R^2 - Entrenamiento

# ==================== MTRICAS PARA DATOS DE VALIDACIN ====================


# Define Xval y Yval para los datos de validacin los cuales se son de la anterior divisin
Xval = nasa.drop('presion', axis=1)
# Estima los valores de Y para los datos de validacin
yhat_val = results.predict(sm.add_constant(Xval))

# Calcula el RSS de validacin
RSS_val = sum((Yval - yhat_val) ** 2)

# Calcula el TSS de validacin
TSS_val = sum((Yval - np.mean(Yval)) ** 2)

# Define el valor de n y m para los datos de validacin
n_val = Xval.shape[0]  # Nmero de muestras
m_val = Xval.shape[1]  # Nmero de predictores

# Calcula el RSE de validacin
RSE_val = np.sqrt(RSS_val / (n_val - m_val - 1))

# Calcula el R^2 de validacin
R2_val = 1 - RSS_val / TSS_val

# Imprimir resultados de validacin
print("Datos de Validacin:")
print(f"Residual Standard Error (RSE):", RSE_val)
print(f"R^2: ", R2_val)
